# 字符串按空格拆分

```c++


#include "bits/stdc++.h"
using namespace std;

int main()
{
    string s1;
    getline(cin, s1);
    vector<string> res;
    while (!s1.empty())
    {
        if (s1.find(" ") == string::npos)
        {
            res.push_back(s1);
            s1.clear();
            break;
        }
        string s_temp = s1.substr(0, s1.find(" "));
        res.push_back(s_temp);
        s1.erase(0, s1.find(" ") + 1);
    }
    for (auto &a : res)
    {
        cout << a << endl;
    }
    return 0;
}
```
